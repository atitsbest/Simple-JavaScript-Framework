1 ﻿Message = new Class ({/**@lends Message*/
  2 
  3 	/**@ignore*/
  4 	Extends: Control,
  5 
  6 	/**@ignore*/
  7 	Implements: Events,
  8 	
  9 	/**
 10 	 * Wird gerade eine persistente Nachricht angezeigt, also eine Nachricht
 11 	 * die zwar überblendet werden kann, aber nach dem Ausblenden der anderen
 12 	 * Nachricht wieder sichtbar wird.
 13 	 * @type Boolean
 14 	 */
 15 	isGlobalFilter: false,
 16 	
 17 	/**
 18 	 * @class Message
 19 	 * @extends Control
 20 	 * @extends Events
 21 	 * @constructs
 22 	 */
 23 	initialize: function(element, options) {
 24 		this.parent(element, options);
 25 	},
 26 	
 27 	/**
 28 	 *
 29 	 */
 30 	onReady: function() {
 31 		this.parent();
 32 	},
 33 	
 34 	/**
 35 	 * Nachricht anzeigen
 36 	 * @param msg Der Text die angezeigt werden soll.
 37 	 * @param type Der Typ der Nachricht z.B.: 'error', 'info', 'succes', ...
 38 	 */
 39 	_showMessage: function(/**String*/msg, /**String*/type, params) {    
 40     //wo soll die Nachricht angezeigt werden?
 41     var $viewElement = this.$element;
 42     if(params && params.viewElement)
 43       $viewElement = $(params.viewElement);
 44     
 45     $viewElement
 46       .html(msg)
 47       .attr('class', 'message ' + type)
 48       .show();
 49       
 50     // Alle Subjekte aufrufen.
 51     this.fireEvent('show');
 52 
 53     if (type == 'info' || type == 'success' || (params && params.timeout)) {
 54       (function() {
 55         this.clear();
 56         // Der Hinweis mit dem GlobalFilter muss erhalten bleiben!
 57         if (this.isGlobalFilter)
 58           this.filter(this.isGlobalFilter);
 59       }).delay(10000, this);
 60     }
 61 	},
 62 	
 63 	/**
 64 	 * Nachricht ausblenden
 65 	 */
 66 	clear: function(params) {
 67     //welche Nachricht soll ausgeblendet werden?
 68     var $viewElement = this.$element;
 69     if(params && params.viewElement)
 70       $viewElement = $(params.viewElement);
 71       
 72     $viewElement
 73         .hide();
 74         
 75     // Alle Subjekte aufrufen.
 76     this.fireEvent('deliver');
 77 	},
 78 	
 79 	
 80 	/**
 81 	 * Nachrichten anzeigen
 82 	 * @param msg Der Text.
 83 	 * @param params Optionen
 84 	 * @param params.type Der Type der Nachricht z.B.: 'error', 'info', 'succes', ...
 85 	 */
 86 	show: function(msg, params) {
 87 	  this._showMessage(msg.Message, msg.type.toString().toLocaleLowerCase(), params);
 88 	},
 89 	
 90 	/**
 91 	 * Einen Fehler anzeigen.
 92 	 * @param msg Der Text.
 93 	 */ 
 94 	error: function(msg, params) {
 95 	  this._showMessage(msg, 'error', params);
 96 	},
 97 	
 98 	/**
 99 	 * Eine Information anzeigen. Wird nach 10sec ausgeblendet.
100 	 * @param msg Der Text.
101 	 */ 
102 	info: function(msg) {    
103 	  this._showMessage(msg, 'info');
104 	},
105 	
106 	/**
107 	 * Einen Erfolg anzeigen. Wird nach 10sec ausgeblendet.
108 	 * @param msg Der Text.
109 	 */ 
110 	success: function(msg) {    
111 	  this._showMessage(msg, 'success');
112 	},
113 	
114 	/**
115 	 * Eine Warnung anzeigen.
116 	 * @param msg Der Text.
117 	 */ 
118 	warning: function(msg) {    
119 	  this._showMessage(msg, 'warning');
120 	},
121 	
122 	/**
123 	 * Eine persistente Nachricht anzeigen {@link Message.isGlobalFilter}.
124 	 * @param msg Der Text.
125 	 */ 
126 	filter: function(msg, params) {    
127 	  this.isGlobalFilter = msg;
128 	  this._showMessage(msg, 'filter', params);  
129 	},
130 	
131 	autoSaveInfo: function(msg, params){
132 	  this._showMessage(msg, 'autoSaveInfo');
133 	}
134 	
135 });
