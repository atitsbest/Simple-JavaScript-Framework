 36 function JSLoad(opt) { //tags, path, version, executeAfterLoad, scriptConcatenatorURL) {
 37   // convert the tags array into a hash and keep a separate tagOrder array.
 38   if (!opt.tags) return;
 39   var objectHash = {};
 40   var ordering = [];
 41   for (var i = 0; i < opt.tags.length; i++) {
 42     var tag = opt.tags[i];
 43     objectHash[tag.name] = tag;
 44     ordering.push(tag.name);
 45   }
 46   this.tags = objectHash;
 47   this.tagOrder = ordering;
 48 
 49   // set flags and one-time data
 50   this.path = opt.path ? opt.path : "";
 51   this.version = opt.version;
 52   this.executeAfterLoad = opt.executeAfterLoad ? true : false;
 53   this.scriptConcatenatorURL = opt.scriptConcatenatorURL;
 54   
 55   // for Safari when using a scriptConcatenator
 56   if (this.scriptConcatenatorURL && /WebKit|khtml/i.test(navigator.userAgent)) {
 57     this.safariSetsLoaded = {}; 
 58   }
 59 
 60   // Initialize "Private" attributes
 61   this.tagsLoaded = {};
 62   this.sourceFileSetQueue = [];
 63   this.sourceFilesLoaded = {};
 64   this.ALLJS = true; // a constant
 65 }

 67 JSLoad.prototype.load = function(tagNames, callback, path, version, executeAfterLoad) {
 68   if (!tagNames) return; // tagNames required
 69   var path = path || this.path;
 70   var version = version || this.version;  
 71   var executeAfterLoad = executeAfterLoad || this.executeAfterLoad;
 72 
 73   // Keep track of all the source files that we are about to load
 74   // This allows us to skip repeat calls to load a file, if it is pending.
 75   var srcToLoad = this.getSrcToLoad(tagNames, path);
 76   this.sourceFileSetQueue.push({
 77     srcToLoad: srcToLoad,
 78     callback: callback,
 79     version: version
 80   });
 81 
 82   // Finally, load all the source files. Only run loadScript if it's
 83   // not running. ExecuteAfterLoad will postpone all loading until after the page
 84   // has loaded.
 85   if (!this.sourceFileSetQueue.isRunning) {
 86     if (executeAfterLoad) {
 87       // Note: this is a dependency on FastInit that should probably be removed in the
 88       // public version of JSLoad. We could copy FastInit code into here, instead.
 89       var thisObj = this;      
 90       FastInit.addOnLoad( function () {
 91         thisObj.loadScript(thisObj.sourceFileSetQueue[0]);
 92       });
 93       // Don't have any more loadScript calls set. The first call after the 
 94       // page load will open this back up.
 95       this.sourceFileSetQueue.isRunning = true; 
 96     } else {
 97       this.loadScript(this.sourceFileSetQueue[0]);
 98     }
 99   }
100 }

102 JSLoad.prototype.getSrcToLoad = function(tagNames, path) {
103   if (!path) var path = this.path;
104 
105   // for closures in markDependentTags()
106   var tags = this.tags;
107   var dependentTags = {};
108   
109   // private function, to be called recursively
110   function markDependentTags(tagName) {
111     var tag = tags[tagName];
112     if (!tag) return;    
113     dependentTags[tagName] = true;      
114     if (!tag.requires) return;
115     for (var i = 0; i < tag.requires.length; i++) {
116       var requiredTagName = tag.requires[i];
117       // only load if not already loaded
118       if (!dependentTags[requiredTagName]) { 
119         markDependentTags(requiredTagName);
120       }
121     }
122   }
123 
124   // create the full set of dependent tags
125   for (var i = 0; i < tagNames.length; i++) {
126     markDependentTags(tagNames[i]);
127   }
128 
129   // Using the tag order, figure out what source files to load.
130   // Don't load a source file if any of the following is true:
131   //  - the tag is not linked to a source file ("tagOnly")
132   //  - the tag isn't marked for loading
133   //  - the tag has an isLoaded test and that test returns true
134   //  - the tag's source file has already been loaded
135   //  - the tag's source file is in the queue for loading
136   var srcToLoad = [];
137   for (var i = 0; i < this.tagOrder.length; i++) {
138     var tagName = this.tagOrder[i];
139     var tag = this.tags[tagName];
140 
141     if (tag.tagOnly ||
142         !dependentTags[tagName] ||
143         (tag.isLoaded && tag.isLoaded())) continue;
144     if ( tagName.indexOf("http://") > -1 ) {
145       var filePath = tagName;
146     } else {
147       var filePath = (path ? path : "") + tagName + '.js';
148     }
149     if (this.sourceFilesLoaded[filePath] || this.isQueued(filePath)) continue;
150 
151     srcToLoad.push(filePath);
152   }
153 
154   return srcToLoad;
155 }

157 JSLoad.prototype.loadScript = function(srcSetObj, iteration) {
158   this.sourceFileSetQueue.isRunning = true;
159   var thisObj = this; // for closures
160   var thisFn = arguments.callee;
161 
162   // This fires the callback when a srcSet is finished loading. First, it
163   // executes the callback.
164   function loadNext () {
165     if (srcSetObj.callback) srcSetObj.callback();
166     thisObj.sourceFileSetQueue.shift();
167     if (thisObj.sourceFileSetQueue.length > 0) {
168       thisFn.call(thisObj, thisObj.sourceFileSetQueue[0]);
169     } else {
170       thisObj.sourceFileSetQueue.isRunning = false;
171     }  
172   }
173   
174   // Creates script el, adds onload handlers, and inserts into the dom.
175   function createScriptEl (url, srcSetObj, iteration) {
176     // REFACTOR!
177     if (url == 'undefined')
178       return;
179   
180     // Create script, set properties, load
181 	  var script = document.createElement("script");
182   	script.type = "text/javascript";
183   	script.src = url
184 	  // script onload, handling Safari 2.0
185     function scriptOnLoad() { 
186   		if (script.readyState && script.readyState != "loaded" && 
187   		    script.readyState != "complete") { return; }
188   		script.onreadystatechange = script.onload = null;
189       // next iteration
190       if (thisObj.scriptConcatenatorURL) {
191         loadNext();
192       } else {
193         thisObj.loadScript(srcSetObj, iteration);
194       }
195   	};
196   	if (thisObj.safariSetsLoaded) { // Safari hack
197       var callbackTimer = setInterval(function() {
198         if (thisObj.safariSetsLoaded[srcSetObj.srcToLoad.join(",")]) {
199           clearInterval(callbackTimer);
200           scriptOnLoad();
201         }
202       }, 100);
203     } else {
204       script.onload = script.onreadystatechange = scriptOnLoad;
205     }
206 	  document.getElementsByTagName("head")[0].appendChild(script);
207   }
208 
209 	// If there are no source files in this set, just execute the callback and
210 	// move on.
211 	if (srcSetObj.srcToLoad.length == 0) {
212 	  loadNext();
213 	  
214 	// If there are source files, run them and set the callbacks to run when
215 	// the source files finish.
216 	} else {
217   	// If we're using a script concatenator on the server, then we load
218   	// all the scripts in one fell swoop.
219   	if (this.scriptConcatenatorURL) {
220   	  // Mark all files as loaded
221     	for (var i = 0; i < srcSetObj.srcToLoad.length; i++) {
222     	  var url = srcSetObj.srcToLoad[i];
223       	this.sourceFilesLoaded[url] = true;
224     	}  	  
225       createScriptEl( 
226         this.scriptConcatenatorURL
227           + "&filePaths="
228           + srcSetObj.srcToLoad.join(",")
229       	  + (srcSetObj.version ? "&version=" + srcSetObj.version : ""),
230       	srcSetObj
231       );
232     
233     // If we're not using a script concatenator, then this function will
234     // recurse through the each of the scripts in this srcSet.
235   	} else {
236   	  // If we've hit the end of this srcSet, run loadNext()
237   	  iteration = iteration || 0;
238   	  if ( (iteration + 1) > srcSetObj.srcToLoad.length ) loadNext();
239   	  
240   	  // Mark this file as loaded
241   	  var url = srcSetObj.srcToLoad[iteration];
242   	  this.sourceFilesLoaded[url] = true;
243   	  createScriptEl( 
244   	    url + (srcSetObj.version ? "?version=" + srcSetObj.version : ""),
245   	    srcSetObj,
246   	    iteration + 1
247   	  );
248   	}
249 	}
250 }

252 JSLoad.prototype.isQueued = function(url) {
253   for (var i = 0; i < this.sourceFileSetQueue.length; i++) {
254     var set = this.sourceFileSetQueue[i];
255     for (var j = 0; j < set.srcToLoad.length; j++) {
256       if (url == set.srcToLoad[j]) return true;
257     }
258   }
259   return false;
260 }
