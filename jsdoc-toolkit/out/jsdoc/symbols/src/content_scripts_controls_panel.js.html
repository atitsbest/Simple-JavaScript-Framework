  3 var Panel = Control.extend(/** @lends Panel# */{
  4   /**
  5    * Gruppiert Child-Controls und bietet einen besseren Zugriff auf sie d.h.
  6    * alle Child-Controls sind direkt über this.<childControlName> ansprechbar.
  7    * Mit dem Parameter [url] kann man angeben von wo das HTML für
  8    * das Panel asynchron geladen werden soll.
  9    * 
 10    * @class Panel
 11    * @augments Control
 12    * @constructs
 13    */
 14   init: function(element, options) {
 15     this._super(element, options);
 16     var that = this;
 17 
 18     // Members:
 19     this.pageCount = 0;
 20     this.currentPage = 1;
 21     
 22     /**
 23      * Wenn true, kann das Panel verschiedene Seiten der angegeben Url 
 24      * vom Server laden. Dabei hängt Panel einfach ein 'page=1' an die 
 25      * Url. Der Server muss dann den richtigen Inhalt zurück liefern.
 26      */
 27     this.pagingEnabled = false; /**bool*/
 28 
 29     /**
 30      * Publisher: Wird ausgeführt, wenn das Panel die angegebene Url
 31      * geladen hat.
 32      */
 33     this.onFinishedLoading = new Publisher(); /**Publisher*/
 34 
 35     // Gibt's was zu laden?
 36     // INFO: Wenn das Panel weiteres HTML laden soll dann schon im CTR
 37     //       damit beim onReady bereits alle neu geladenen Controls
 38     //       in der DOM sind und initialisiert werden können.
 39     if (this.url) {
 40       // Paging aktiviert? Wenn ja, dann müssen wir erstmal ermitteln, 
 41       // wieviele Seiten es geben wird.
 42       if (this.pagingEnabled) {
 43         var url = $.extend(url, this.url, {params:{getpagecount:true}});
 44         // Die Anzahl der Seite muss vom Server ermittelt werden.
 45         $.getJSON(mvc.buildUrl(url), function(data) {
 46           that.pageCount = data.result;
 47           that.load();
 48         });
 49       }
 50       else {
 51         this.load();
 52       }
 53     }
 54   },
 55 
 56   /**
 57    * Controlbaum ist aufgebaut.
 58    * @param callback Wird aufgerufen, wenn onReady fertig ist.
 59    */
 60   onReady: function (/**function*/callback) {
 61     var that = this;
 62 
 63     // Alle Kindercontrols als Properties bereitstellen.
 64     this.controls.forEach(function(ctrl) {
 65        that[ctrl.id] = ctrl;
 66     });
 67     
 68     this._super(callback);
 69   },
 70 
 71   /**
 72    * Fügt den Content von url in $element ein.
 73    * @param url=this.url Von hier kommt der Content. Sollte url leer sein wird die url aus den options verwendet.
 74    */
 75   load: function(/**string*/url) {
 76     var that = this;
 77     url = url || this.url;
 78     
 79     // Müssen wir blätter?
 80     if (this.pagingEnabled) {
 81       // Die Url um die Paging-Infos erweitern.
 82       url = $.extend(url, {params: {page: this.currentPage}});
 83     }
 84 
 85     // Element mit einer Nachricht blockieren. Die Nachricht (loadMessage) kann 
 86     // mit den options für das Panel übergeben werden 
 87     // z.B.: <div class="{ type : 'Panel', loadMessage: 'Bitte warten', ...}">...
 88     this.$element.block(this.loadMessage);
 89     
 90     // Inhalt für das Panel laden.
 91     this.$element.load(mvc.buildUrl(url), function() {
 92       // Controls aufbauen.
 93       // INFO: Im neu geladenen HTML können Controls sein, die
 94       //       initialisiert werden wollen.
 95       that.initializeChildControls();
 96       that.onReady(function() {
 97         that.$element.unblock();
 98         // Benachrichtigen.
 99         that.onFinishedLoading.deliver(that);
100       });
101     });
102   }
103 });
