 14 /**
 15  * Nachricht anzeigen
 16  * @controller
 17  * @action
 18  * @parameter (als hash)
 19  * oder: { controller, action, parameter }
 20  *
 21  * INFO: Parameter können als auch Funktionen angegeben werden,
 22  *       die jedes mal beim Zusammenstellen der Url ausgewertet werden.
 23  *       Auf diese Weise lässt sich im Markup eine Url angeben die
 24  *       z.B.: den jeweils ausgewählen Wert eines <select> verwendet.
 25  */
 26 mvc.buildUrl = function() {    
 27   var args = mvc.buildUrl.arguments;
 28   
 29   // Wurde ein Hash übergeben, oder ein string
 30   if (args.length == 1) {
 31     // Der erste Parameter ist ein Hash mit { controller:, action:, params: }
 32     var url = args[0];
 33     return mvc.buildUrl(url.controller, url.action, url.params, url.noCache);
 34   }
 35   else {
 36     var result = "";
 37 
 38     if (args[0])
 39       result = mvc.baseUrl + args[0] + '/';
 40     if (args[1])
 41       result += args[1] + mvc.actionExtension;
 42     
 43     if (args[2]) {
 44       if (typeof args[2]  == 'string') {
 45         result += args[2];
 46       }
 47       else {
 48         // Aus dem params-Objekt eine string bauen.
 49         var counter = 0;
 50         for (var param in args[2]) { 
 51           result += (counter++ ? '&' : '?') + param + '=';
 52           // Ist der Parameter eine Function?
 53           result += ($.isFunction(args[2][param]))
 54             ? args[2][param]().toString()
 55             : args[2][param];
 56         }
 57       }
 58     }
 59         
 60     // Zufallsgenerierten Parameter der den Browser am chachen hindert (hindern soll ;)
 61     if (args[3] == undefined || !args[3])
 62       result += (result.lastIndexOf('?') == -1 ? "?" : "&") + "_=" + (String.prototype.rand ? new String().rand(10) : '');
 63     
 64     return encodeURI(result);  
 65   }
 66 };

 68 /**
 69  * Ruft eine Action auf und liefert das Ergebnis
 70  * als JS-Object zurück (JSON)
 71  * @actionName
 72  * @params
 73  * @noinvoke Rückgabewert von HTTP-Request wird nicht als JSON ausgewertet.
 74  */
 75 mvc.invokeRemote = function(actionName, params, noinvoke) {
 76   var url = actionName + mvc.actionExtension;
 77   
 78   // Asynchronen Aufruf starten.
 79   var result = jQuery.ajax({
 80     type: 'GET',
 81     url: url,
 82     data: params,
 83     async: false,
 84     cache: false,
 85     dataType: 'text'});
 86 
 87   // Fehler?
 88   if (result.status != 200)
 89     throw new Error(result.responseText);
 90   else if (!noinvoke) {
 91     // Rückgabewert als JSON  
 92     try {
 93       return eval('(' + result.responseText + ')');
 94     }
 95     catch(e) { alert('Beim Verarbeiten des Rückgabewerts von Aufruf ' + url + ' ist folgender Fehler aufgetreten: ' + e); }
 96   }
 97 };
