162             $.unblockUI(); $('.blockUI').remove();
163             
164             // Fehlermeldung anzeigen?
165             if (settings.dataType == 'json') {

168             }
169             // Hier haben wir es mit einem ungew�nschten Fehler zu tun.

171               $.blockUI(request.responseTest);
172             }
173             break;
174         }
175 
176         // F�r den Fall, dass es von einer Komponente vergessen wurde.
